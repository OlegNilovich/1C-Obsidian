Пример правильного удаления элементов массива по условию. Удалять нужно с конца.

```bsl

МассивЛет = Новый Массив;
МассивЛет.Добавить("2000");
МассивЛет.Добавить("2001");
МассивЛет.Добавить("2002");
МассивЛет.Добавить("2003");
МассивЛет.Добавить("2004");
МассивЛет.Добавить("2005");
МассивЛет.Добавить("2006");
МассивЛет.Добавить("2007");

// Удаляем элементы массива с конца по условию
Индекс = МассивЛет.ВГраница(); // Индексация начинается с 7
Пока Индекс >= 0 Цикл
    Если МассивЛет[Индекс] > "2004" Тогда
        МассивЛет.Удалить(Индекс); // Удаляем по индексу
    КонецЕсли;
    Индекс = Индекс - 1; // Уменьшаем индекс на единицу
КонецЦикла;

// Выводим оставшиеся элементы
Для каждого Число Из МассивЛет Цикл
    Сообщить(Число);
КонецЦикла;
```
