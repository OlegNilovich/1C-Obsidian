Асинх Процедура ПереместитьФайл(Команда)
	
	Файл = Новый Файл(ПеремещаемыйФайл);
	
	Если ЖДАТЬ Файл.СуществуетАсинх() Тогда
		
		Проводник = Новый ДиалогВыбораФайла(РежимДиалогаВыбораФайла.ВыборКаталога);
		Проводник.Заголовок = "Выберите куда переместить файл";
		
		Каталоги = Ждать Проводник.ВыбратьАсинх();
		Если Каталоги = Неопределено Тогда
			Возврат;
		КонецЕсли;
		
		Результат = ЖДАТЬ ПереместитьФайлАсинх(ПеремещаемыйФайл, Каталоги[0] + "\" + Файл.Имя);
		Сообщить("Файл успешно перемещен по пути: " + Результат);
		
	Иначе
		Сообщить("Такого файла не существует!");			
	КонецЕсли;
	
КонецПроцедуры
