###### Первый способ: объект "ТекстовыйДокумент"
```bsl
&НаКлиенте
Асинх Процедура Загрузить_CSV(Команда)
	
	Товары.Очистить();
	
	Текст = Новый ТекстовыйДокумент;
	Ждать Текст.ПрочитатьАсинх(ФайлЗагрузки, КодировкаТекста.UTF8);
	
	Для НомерСтроки = 2 По Текст.КоличествоСтрок() Цикл
		
		ТекСтрока = Текст.ПолучитьСтроку(НомерСтроки);
		ЧастиСтроки = СтрРазделить(ТекСтрока, ";");
		
		НоваяСтрока = Товары.Добавить();
		НоваяСтрока.Наименование			= ЧастиСтроки[0];
		НоваяСтрока.ВидНоменклатуры	= ЧастиСтроки[1]; 
		НоваяСтрока.Артикул					= ЧастиСтроки[2];
		НоваяСтрока.ЕдиницаИзмерения	= ЧастиСтроки[3];
		НоваяСтрока.СтавкаНДС				= ЧастиСтроки[4];
		
	КонецЦикла;  
	
КонецПроцедуры
```

###### Второй способ: объект "ЧтениеТекста"
```bsl
&НаКлиенте
Асинх Процедура Загрузить_CSV(Команда)
	
	Товары.Очистить();
	
	Текст = Новый ЧтениеТекста;
	Текст.Открыть(ФайлЗагрузки);
	
	ТекСтрока = Текст.ПрочитатьСтроку();
	НомерСтроки = 2;
	
	Пока ТекСтрока <> Неопределено Цикл
		
		ЧастиСтроки = СтрРазделить(ТекСтрока, ";");
		
		НоваяСтрока = Товары.Добавить();
		НоваяСтрока.Наименование			= ЧастиСтроки[0];
		НоваяСтрока.ВидНоменклатуры	= ЧастиСтроки[1]; 
		НоваяСтрока.Артикул					= ЧастиСтроки[2];
		НоваяСтрока.ЕдиницаИзмерения	= ЧастиСтроки[3];
		НоваяСтрока.СтавкаНДС				= ЧастиСтроки[4];
		
		ТекСтрока = Текст.ПрочитатьСтроку();  
		НомерСтроки = НомерСтроки + 1;
	
	КонецЦикла;
	
КонецПроцедуры
```
