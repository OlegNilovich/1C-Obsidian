###### Первый способ: объект "ФабрикаXDTO"
```bsl
&НаКлиенте
Процедура Загрузить_XML(Команда)
	
	Товары.Очистить();
	
	ЧтениеXML = Новый ЧтениеXML;
	ЧтениеXML.ОткрытьФайл(ФайлЗагрузки);

	СписокТоваров = ФабрикаXDTO.ПрочитатьXML(ЧтениеXML).Товар;
	
	Если ТипЗнч(СписокТоваров) = Тип("ОбъектXDTO") Тогда
		НоваяСтрока = Товары.Добавить();
		НоваяСтрока.Наименование			= СписокТоваров.Наименование;
		НоваяСтрока.ВидНоменклатуры	= СписокТоваров.ВидНоменклатуры;
		НоваяСтрока.Артикул					= СписокТоваров.Артикул;
		НоваяСтрока.ЕдиницаИзмерения	= СписокТоваров.ЕдиницаИзмерения;
		НоваяСтрока.СтавкаНДС				= СписокТоваров.СтавкаНДС;
		Возврат;
	КонецЕсли;

	Для каждого ТоварОбъект Из СписокТоваров Цикл
		НоваяСтрока = Товары.Добавить();

		НоваяСтрока.СтавкаНДС				= ТоварОбъект.СтавкаНДС;

КонецПроцедуры
```

###### Второй способ: объект "ПостроительDOM"
```bsl
&НаКлиенте
Процедура Загрузить_XML(Команда)

	Товары.Очистить();

	ЧтениеXML = Новый ЧтениеXML;
	ЧтениеXML.ОткрытьФайл(ФайлЗагрузки);
	ПостроительDOM = Новый ПостроительDOM;
	ДокументDOM = ПостроительDOM.Прочитать(ЧтениеXML);

	СписокТоваров= ДокументDOM.ЭлементДокумента;
	
	Для каждого Товар Из СписокТоваров.ДочерниеУзлы Цикл
		
		Если (Товар.ТипУзла <> ТипУзлаDOM.Элемент) И (Товар.ИмяУзла <> "Товар")  Тогда
			Продолжить;		
		КонецЕсли;
		
		НоваяСтрока = Товары.Добавить();
		
		Для каждого Свойство Из Товар.ДочерниеУзлы Цикл
			Если Свойство.ИмяУзла = "Наименование" Тогда
				НоваяСтрока.Наименование = Свойство.ТекстовоеСодержимое;
			ИначеЕсли Свойство.ИмяУзла = "ВидНоменклатуры" Тогда
				НоваяСтрока.ВидНоменклатуры = Свойство.ТекстовоеСодержимое;
			ИначеЕсли Свойство.ИмяУзла = "Артикул" Тогда
				НоваяСтрока.Артикул = Свойство.ТекстовоеСодержимое;
			ИначеЕсли Свойство.ИмяУзла = "ЕдиницаИзмерения" Тогда
				НоваяСтрока.ЕдиницаИзмерения = Свойство.ТекстовоеСодержимое;
			ИначеЕсли Свойство.ИмяУзла = "СтавкаНДС" Тогда
				НоваяСтрока.СтавкаНДС = Свойство.ТекстовоеСодержимое;
			КонецЕсли;
		КонецЦикла;
		
	КонецЦикла; 
	
	ЧтениеXML.Закрыть();
	
КонецПроцедуры
```
###### Третий способ: объект "ЧтениеXML"
```bsl
Процедура Загрузить_XML(Команда)

	Товары.Очистить();
	
	ЧтениеXML = Новый ЧтениеXML;
	ЧтениеXML.ОткрытьФайл(ФайлЗагрузки);
	
	ИмяТекущегоУзла = "";
	
	Пока ЧтениеXML.Прочитать() Цикл
		
		Если ЧтениеXML.ТипУзла = ТипУзлаXML.НачалоЭлемента Тогда
			
			Если ЧтениеXML.Имя = "Товар" Тогда
				НоваяСтрока = Товары.Добавить();	
			КонецЕсли;
			
			ИмяТекущегоУзла = ЧтениеXML.Имя;

		ИначеЕсли ЧтениеXML.ТипУзла = ТипУзлаXML.Текст Тогда
		
			Если ИмяТекущегоУзла = "Наименование" Тогда
				НоваяСтрока.Наименование = ЧтениеXML.Значение;
			ИначеЕсли Свойство.ИмяУзла = "ВидНоменклатуры" Тогда
				НоваяСтрока.ВидНоменклатуры = ЧтениеXML.Значение;
			ИначеЕсли Свойство.ИмяУзла = "Артикул" Тогда
				НоваяСтрока.Артикул = ЧтениеXML.Значение;
			ИначеЕсли Свойство.ИмяУзла = "ЕдиницаИзмерения" Тогда
				НоваяСтрока.ЕдиницаИзмерения = ЧтениеXML.Значение;
			ИначеЕсли Свойство.ИмяУзла = "СтавкаНДС" Тогда
				НоваяСтрока.СтавкаНДС = ЧтениеXML.Значение;
			КонецЕсли;
			
		КонецЕсли;
		
	КонецЦикла;   
	
	ЧтениеXML.Закрыть();
	
КонецПроцедуры
```
