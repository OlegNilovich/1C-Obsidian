Процедура Создать_DBF(Команда)
	
	ФайлДБФ = Новый XBase; 
	ФайлДБФ.Поля.Добавить("Name", "S", 100);
	ФайлДБФ.Поля.Добавить("Type", "S", 50); 
	ФайлДБФ.Поля.Добавить("Article", "S", 20); 
	ФайлДБФ.Поля.Добавить("Unit", "S", 10); 
	ФайлДБФ.Поля.Добавить("VatRate", "S", 5); 

	ФайлДБФ.СоздатьФайл(МестоВыгрузки + "\" + "DBF-Файл"  + ".dbf");
	ФайлДБФ.АвтоСохранение = Истина;
	
	Для каждого Товар Из Товары Цикл
		ФайлДБФ.Добавить();
		ФайлДБФ.Name	= Товар.Наименование;
		ФайлДБФ.Type		= Строка(Товар.ВидНоменклатуры);
		ФайлДБФ.Article	= Товар.Артикул;
		ФайлДБФ.Unit		= Строка(Товар.ЕдиницаИзмерения);
		ФайлДБФ.VatRate	= Строка(Товар.СтавкаНДС);
		ФайлДБФ.Записать();	
	КонецЦикла;
	
	ФайлДБФ.ЗакрытьФайл();
	ПоказатьОповещениеПользователя("Данные выгружены успешно!");
	ЗапуститьПриложение(МестоВыгрузки);
	
КонецПроцедуры
