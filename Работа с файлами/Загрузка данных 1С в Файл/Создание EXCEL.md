###### Первый способ: По ячейкам
```bsl
&НаКлиенте
Асинх Процедура Создать_EXCEL(Команда)
	
	ЦветЖелтый = Новый Цвет(255,255,0);
	
	ТабДок = Новый ТабличныйДокумент;
	
	ТабДок.Область(1,1).Текст = "Наименование";	
	ТабДок.Область(1,1).ЦветФона = ЦветЖелтый; 
	ТабДок.Область(1,1).ШиринаКолонки = 32;
	
	ТабДок.Область(1,2).Текст = "Вид номенклатуры";
	ТабДок.Область(1,2).ЦветФона = ЦветЖелтый;
	ТабДок.Область(1,2).ШиринаКолонки = 16;
	
	ТабДок.Область(1,3).Текст = "Артикул";        
	ТабДок.Область(1,3).ЦветФона = ЦветЖелтый;
	ТабДок.Область(1,3).ШиринаКолонки = 12;
	
	ТабДок.Область(1,4).Текст = "ЕдиницаИзмерения";        
	ТабДок.Область(1,4).ЦветФона = ЦветЖелтый;
	ТабДок.Область(1,4).ШиринаКолонки = 16;
	
	ТабДок.Область(1,5).Текст = "СтавкаНДС";        
	ТабДок.Область(1,5).ЦветФона = ЦветЖелтый;
	ТабДок.Область(1,5).ШиринаКолонки = 12;
	
	НомерСтроки = 2;
	Для каждого Товар Из Товары Цикл
		ТабДок.Область(НомерСтроки, 1).Текст = Товар.Наименование;        
		ТабДок.Область(НомерСтроки, 2).Текст = Строка(Товар.ВидНоменклатуры);        
		ТабДок.Область(НомерСтроки, 3).Текст = Товар.Артикул;        
		ТабДок.Область(НомерСтроки, 4).Текст = Строка(Товар.ЕдиницаИзмерения);        
		ТабДок.Область(НомерСтроки, 5).Текст = Строка(Товар.СтавкаНДС);        
		НомерСтроки = НомерСтроки + 1;
	КонецЦикла;
	
	ПутьКФайлу = МестоВыгрузки + "\" + "EXCEL-Файл"  + ".xlsx";
	ЗаписьУспешна = Ждать ТабДок.ЗаписатьАсинх(ПутьКФайлу, ТипФайлаТабличногоДокумента.XLSX);
	Если ЗаписьУспешна Тогда
		ПоказатьОповещениеПользователя("Данные выгружены успешно!");
		ЗапуститьПриложение(МестоВыгрузки);
	КонецЕсли;
	
КонецПроцедуры
```
###### Второй способ: по макету
```bsl
&НаКлиенте
Асинх Процедура Создать_EXCEL(Команда)
		
	ТабДок = Новый ТабличныйДокумент;
	
	ЗаполнитьТабличныйДокументНаСервере(ТабДок);
	
	ЗаписьУспешна = Ждать ТабДок.ЗаписатьАсинх(ПутьКФайлу, ТипФайлаТабличногоДокумента.XLSX);
	
	Если ЗаписьУспешна Тогда
		ПоказатьОповещениеПользователя("Данные выгружены успешно!");
	КонецЕсли;

КонецПроцедуры

&НаСервере
Процедура ЗаполнитьТабличныйДокументНаСервере(ТабДок)
	
	Макет = Обработки.ВыгрузкаДанныхВФайлы.ПолучитьМакет("МакетВыгрузкиВExcel");
	
	ОбластьШапка = Макет.ПолучитьОбласть("ШапкаТаблицы");
	ТабДок.Вывести(ОбластьШапка);
	
	Для каждого СтрокаТЧ Из Объект.Данные1С Цикл
		ОбластьСтрока = Макет.ПолучитьОбласть("СтрокаТаблицы");	
		ОбластьСтрока.Параметры.Заполнить(СтрокаТЧ);
		ТабДок.Вывести(ОбластьСтрока);
	КонецЦикла;

КонецПроцедуры
```
