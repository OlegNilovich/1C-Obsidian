###### Первый способ: глобальная процедура ЗаписатьJSON()
```bsl
&НаКлиенте
Процедура Создать_JSON(Команда)
	
	МассивКонтрагентов = Новый Массив;
	
	Для каждого Товар Из Товары Цикл
		СтруктураКонтрагента = Новый Структура;
		СтруктураКонтрагента.Вставить("Наименование", Товар.Наименование);
		СтруктураКонтрагента.Вставить("ВидНоменклатуры", Строка(Товар.ВидНоменклатуры));
		СтруктураКонтрагента.Вставить("Артикул", Товар.Артикул);   
		СтруктураКонтрагента.Вставить("ЕдиницаИзмерения", Строка(Товар.ЕдиницаИзмерения));   
		СтруктураКонтрагента.Вставить("СтавкаНДС", Строка(Товар.СтавкаНДС));   
		МассивКонтрагентов.Добавить(СтруктураКонтрагента);
	КонецЦикла;                                               
	
	ПутьКФайлу = МестоВыгрузки + "\" + "JSON-Файл"  + ".json";
	ЗаписьJSON = Новый ЗаписьJSON;
	ЗаписьJSON.ОткрытьФайл(ПутьКФайлу,,, Новый ПараметрыЗаписиJSON(, Символы.Таб));
	ЗаписатьJSON(ЗаписьJSON, МассивКонтрагентов); 
	ЗаписьJSON.Закрыть();
	
	ПоказатьОповещениеПользователя("Данные выгружены успешно!");
	ЗапуститьПриложение(МестоВыгрузки);
	
КонецПроцедуры
```
###### Второй способ: метод объекта "ЗаписьJSON"
```bsl
&НаКлиенте
Процедура Создать_JSON(Команда)

	ПутьКФайлу = МестоВыгрузки + "\" + "JSON-Файл"  + ".json";
	ЗаписьJSON = Новый ЗаписьJSON;
	ЗаписьJSON.ОткрытьФайл(ПутьКФайлу,,, Новый ПараметрыЗаписиJSON(, Символы.Таб));

	ЗаписьJSON.ЗаписатьНачалоМассива();

	Для каждого Товар Из Товары Цикл
		
		ЗаписьJSON.ЗаписатьНачалоОбъекта();

			ЗаписьJSON.ЗаписатьИмяСвойства("Наименование");
			ЗаписьJSON.ЗаписатьЗначение(Товар.Наименование);

			ЗаписьJSON.ЗаписатьИмяСвойства("ВидНоменклатуры");
			ЗаписьJSON.ЗаписатьЗначение(Строка(Товар.ВидНоменклатуры));
	
			ЗаписьJSON.ЗаписатьИмяСвойства("Артикул");
			ЗаписьJSON.ЗаписатьЗначение(Товар.Артикул);


			ЗаписьJSON.ЗаписатьИмяСвойства("ЕдиницаИзмерения");
			ЗаписьJSON.ЗаписатьЗначение(Строка(Товар.ЕдиницаИзмерения));
		
			ЗаписьJSON.ЗаписатьИмяСвойства("СтавкаНДС");
			ЗаписьJSON.ЗаписатьЗначение(Строка(Товар.СтавкаНДС));
			
		ЗаписьJSON.ЗаписатьКонецОбъекта();
	
	КонецЦикла;   
	
	ЗаписьJSON.ЗаписатьКонецМассива();
	
	ЗаписьJSON.Закрыть();
	ПоказатьОповещениеПользователя("Данные выгружены успешно!");
	ЗапуститьПриложение(МестоВыгрузки);
	
КонецПроцедуры
```
