```bsl
Процедура УдалитьПовторяющиесяСимволы(ИсходнаяСтрока)
	
	// Новая строка для хранения уникальных символов
	УникальныеСимволы = "";
	
	// Проходим по каждому символу исходной строки
	Для Индекс = 1 По СтрДлина(ИсходнаяСтрока) Цикл
		
		Символ = Сред(ИсходнаяСтрока, Индекс, 1);
		
		// Если символ не пробел и его еще нет в новой строке
		Если Символ <> " " И Найти(УникальныеСимволы, Символ) = 0 Тогда
			УникальныеСимволы = УникальныеСимволы + Символ;
		КонецЕсли;
		
	КонецЦикла;
	
	// Выводим результат
	Сообщить(УникальныеСимволы);
	
КонецПроцедуры

// Пример вызова процедуры
УдалитьПовторяющиесяСимволы("abc cde def");
```
