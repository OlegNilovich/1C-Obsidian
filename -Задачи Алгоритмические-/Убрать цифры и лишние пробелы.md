Процедура УбратьЦифрыИЛишниеПробелы(ИсходнаяСтрока)
	
	СтрокаБезЦифр = "";
	
	// Удаляем цифры
	Для Индекс = 1 По СтрДлина(ИсходнаяСтрока) Цикл
		Символ = Сред(ИсходнаяСтрока, Индекс, 1);
		
		// Если символ не является цифрой, добавляем его к новой строке
		Если Не (Символ >= "0" И Символ <= "9") Тогда
			СтрокаБезЦифр = СтрокаБезЦифр + Символ;
		КонецЕсли;
	КонецЦикла;
	
	// Убираем пробелы в начале и в конце строки
	СтрокаБезЦифр = СокрЛП(СтрокаБезЦифр);
	
	// Заменяем любые последовательности пробелов на один пробел
	Пока Найти(СтрокаБезЦифр, "  ") > 0 Цикл
		СтрокаБезЦифр = СтрЗаменить(СтрокаБезЦифр, "  ", " ");
	КонецЦикла;
	
	// Выводим результат
	Сообщить(СтрокаБезЦифр);
	
КонецПроцедуры

// Пример вызова процедуры
УбратьЦифрыИЛишниеПробелы("Это          стро4ка   с  8   цифра123ми   7");
