```bsl
// Создаем команды для группировки на форме списка документа "Продажи"

&НаКлиенте
Процедура ГруппировкаПоТипуЦены(Команда)
	
	Компоновщик = Список.КомпоновщикНастроек;
	
	Настройки = Компоновщик.Настройки;
	
	СтруктураНастроек = Настройки.Структура;
	
	СтруктураНастроек.Очистить();
	ГруппировкаДанных = СтруктураНастроек.Добавить(Тип("ГруппировкаКомпоновкиДанных"));
	ГруппировкаДанных.ИдентификаторПользовательскойНастройки = "ГруппировкаПоТипуЦены";
	ГруппировкаДанных.РежимОтображения = РежимОтображенияЭлементаНастройкиКомпоновкиДанных.Недоступный;

	ПоляГруппировки = ГруппировкаДанных.ПоляГруппировки;

	ЭлементГруппировкиДанных = ПоляГруппировки.Элементы.Добавить(Тип("ПолеГруппировкиКомпоновкиДанных"));
	ЭлементГруппировкиДанных.Поле = Новый ПолеКомпоновкиДанных("ТипЦены");
	ЭлементГруппировкиДанных.Использование = Истина;

КонецПроцедуры


&НаКлиенте
Процедура СнятьГруппировки(Команда)
	Список.КомпоновщикНастроек.Настройки.Структура.Очистить();
КонецПроцедуры
```
